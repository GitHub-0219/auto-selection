## 三、应用层监控

### 3.1 健康检查端点

在应用中实现 `/health` 端点：

```javascript
// routes/health.js
const express = require('express');
const router = express.Router();

router.get('/health', async (req, res) => {
  const health = {
    status: 'ok',
    timestamp: new Date().toISOString(),
    uptime: process.uptime(),
    services: {}
  };
  
  // 检查数据库
  try {
    await db.query('SELECT 1');
    health.services.database = 'ok';
  } catch (err) {
    health.services.database = 'error';
    health.status = 'degraded';
  }
  
  // 检查Redis
  try {
    await redis.ping();
    health.services.redis = 'ok';
  } catch (err) {
    health.services.redis = 'error';
    health.status = 'degraded';
  }
  
  const statusCode = health.status === 'ok' ? 200 : 503;
  res.status(statusCode).json(health);
});

module.exports = router;
```

### 3.2 自定义监控指标

使用 `prom-client` 暴露Prometheus格式指标：

```javascript
const client = require('prom-client');

// 创建registry
const register = new client.Registry();
client.collectDefaultMetrics({ register });

// 自定义指标
const httpRequestDuration = new client.Histogram({
  name: 'http_request_duration_seconds',
  help: 'Duration of HTTP requests in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.1, 0.3, 0.5, 0.7, 1, 3, 5, 10]
});
register.registerMetric(httpRequestDuration);

// API响应时间
app.use((req, res, next) => {
  const start = process.hrtime();
  res.on('finish', () => {
    const duration = process.hrtime(start);
    httpRequestDuration.observe(
      { method: req.method, route: req.route, status_code: res.statusCode },
      duration[0] + duration[1] / 1e9
    );
  });
  next();
});

// 指标端点
app.get('/metrics', async (req, res) => {
  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
});
```

## 四、日志管理

### 4.1 Nginx访问日志配置

```nginx
# /etc/nginx/nginx.conf 或 /etc/nginx/conf.d/auto-selection.conf

log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                '$status $body_bytes_sent "$http_referer" '
                '"$http_user_agent" "$http_x_forwarded_for" '
                '$request_time $upstream_response_time';

access_log /var/log/nginx/access.log main;

# 条件日志（关闭健康检查日志）
map $request_uri $loggable {
    ~*^/health$ 0;
    default 1;
}

access_log /var/log/nginx/access.log main if=$loggable;
```

### 4.2 日志分析脚本

```bash
#!/bin/bash
# analyze-logs.sh - 日志分析脚本

LOG_FILE="/var/log/nginx/access.log"
ERROR_LOG="/var/log/nginx/error.log"
PM2_LOG="/var/www/.pm2/logs/auto-selection-error.log"

echo "=================================="
echo "日志分析报告 - $(date)"
echo "=================================="

echo ""
echo "【1. 访问统计】"
echo "总请求数: $(wc -l < $LOG_FILE)"
echo "今日请求: $(grep $(date +%d/%b/%Y) $LOG_FILE | wc -l)"

echo ""
echo "【2. HTTP状态码分布】"
awk '{print $9}' $LOG_FILE | sort | uniq -c | sort -rn

echo ""
echo "【3. Top 20 请求IP】"
awk '{print $1}' $LOG_FILE | sort | uniq -c | sort -rn | head -20

echo ""
echo "【4. 慢请求 (>5s)】"
awk '$NF > 5 {print $NF" "$7}' $LOG_FILE | sort -rn | head -10

echo ""
echo "【5. 错误日志】"
echo "最近10条错误:"
tail -n 10 $ERROR_LOG

echo ""
echo "【6. PM2应用错误】"
echo "最近10条应用错误:"
tail -n 10 $PM2_LOG

echo ""
echo "=================================="
```

## 七、自动化告警处理

### 7.1 阿里云OOS自动化运维

创建自动清理任务：

```yaml
名称: Auto选品-磁盘自动清理
触发条件: 磁盘使用率 > 85%
执行操作:
  - 清理PM2日志: find ~/.pm2/logs -mtime +7 -delete
  - 清理Nginx日志: find /var/log/nginx -mtime +7 -name "*.log" -exec truncate -s 0 {} \;
  - 清理临时文件: find /tmp -mtime +1 -type f -delete
```

### 7.2 自动重启脚本

```bash
#!/bin/bash
# auto-restart.sh - 内存超限自动重启

MEM_LIMIT=450
CURRENT_MEM=$(pm2 jlist | jq '.[] | select(.name=="auto-selection") | .monit.memory' | head -1)
LIMIT=$((MEM_LIMIT * 1024 * 1024))

if [ "$CURRENT_MEM" -gt "$LIMIT" ]; then
    echo "$(date): Memory exceeded limit ($CURRENT_MEM > $LIMIT), restarting..."
    pm2 restart auto-selection
fi
```
