import { IsString, IsNotEmpty, Length, validateSync } from 'class-validator';

  @IsNotEmpty()
  @Length(32, 128, { message: 'JWT_SECRET must be at least 32 characters' })
  JWT_SECRET: string;

  @IsString()
  @IsNotEmpty()

  @IsNotEmpty()
  JWT_REFRESH_EXPIRES_IN: string;
}

  const validatedConfig = plainToClass(EnvironmentVariables, config, {
    enableImplicitConversion: true,
  });
  const errors = validateSync(validatedConfig, { skipMissingProperties: false });
  if (errors.length > 0) {
    throw new Error(`Environment validation failed: ${errors.toString()}`);
  }
  return validatedConfig;
}

// src/modules/auth/account-lock.service.ts
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { RedisService } from '../redis/redis.service';

@Injectable()
export class AccountLockService {
  private readonly LOCK_DURATION = 15 * 60; // 15分钟
  private readonly MAX_ATTEMPTS = 5;
  private readonly KEY_PREFIX = 'login:attempts:';

  constructor(
    private prisma: PrismaService,
    private redis: RedisService,
  ) {}

  async recordFailedAttempt(email: string, ip: string): Promise<boolean> {
    const key = `${this.KEY_PREFIX}${email}:${ip}`;
    const attempts = await this.redis.incr(key);
    await this.redis.expire(key, this.LOCK_DURATION);
    
    if (attempts >= this.MAX_ATTEMPTS) {
      await this.lockAccount(email, ip);
      return true; // 已锁定
    }
    return false;
  }

  async isAccountLocked(email: string, ip: string): Promise<boolean> {
    const key = `${this.KEY_PREFIX}${email}:${ip}`;
    const attempts = await this.redis.get(key);
    return parseInt(attempts || '0') >= this.MAX_ATTEMPTS;
  }

  async clearFailedAttempts(email: string, ip: string): Promise<void> {
    const key = `${this.KEY_PREFIX}${email}:${ip}`;
    await this.redis.del(key);
  }

  private async lockAccount(email: string, ip: string): Promise<void> {
    // 可选：发送安全通知邮件
    const user = await this.prisma.user.findUnique({ where: { email } });
    if (user) {
      // await this.mailService.sendSecurityAlert(user.email);
    }
  }
}

    throw error;
  }
}

// 方案B: 滑动验证码 (适合国内用户)
import { RedisService } from '../redis/redis.service';

@Injectable()
export class SlideCaptchaService {
  constructor(private redis: RedisService) {}

  // 生成验证码
  async generateCaptcha(scene: string): Promise<{
    captchaId: string;
    dataURL: string; // 验证码图片
    backgroundImage: string;
    sliderImage: string;
  }> {
    const captchaId = crypto.randomUUID();
    // 生成随机滑块位置 (10-80%)
    const x = Math.floor(Math.random() * 70) + 10;
    const y = Math.floor(Math.random() * 40) + 30;
    
    await this.redis.setex(`captcha:${captchaId}`, 300, JSON.stringify({ x, y, scene }));
    
    return {
      captchaId,
      dataURL: '', // 生成验证码图片的base64
      backgroundImage: '', // 背景图
      sliderImage: '', // 滑块图
    };
  }

  // 验证滑块
  async verifyCaptcha(captchaId: string, trackData: number[]): Promise<boolean> {
    const cached = await this.redis.get(`captcha:${captchaId}`);
    if (!cached) return false;
    
    const { x, y, scene } = JSON.parse(cached);
    const userX = this.calculateEndPosition(trackData);
    
    // 允许±3像素误差
    const tolerance = 3;
    const passed = Math.abs(userX - x) <= tolerance;
    
    if (passed) {
      await this.redis.del(`captcha:${captchaId}`);
    }
    
    return passed;
  }

  private calculateEndPosition(trackData: number[]): number {
    // 根据滑动轨迹计算终点位置
    if (trackData.length === 0) return 0;
    return trackData[trackData.length - 1];
  }
}

@Injectable()
export class AuthService {
  constructor(private redis: RedisService) {}

  async getTodayRegisterCount(ip: string): Promise<number> {
    const key = `register:ip:${ip}:${this.getToday()}`;
    const count = await this.redis.get(key);
    return parseInt(count || '0');
  }

  async incrementRegisterCount(ip: string): Promise<void> {
    const key = `register:ip:${ip}:${this.getToday()}`;
    await this.redis.incr(key);
    await this.redis.expire(key, 86400); // 24小时过期
  }

  private getToday(): string {
    return new Date().toISOString().split('T')[0];
  }
}

// 1. 创建资源所有权守卫
// src/common/decorators/resource-owner.decorator.ts
import { SetMetadata } from '@nestjs/common';

export const RESOURCE_OWNER_KEY = 'resource_owner';
export const ResourceOwner = (resourceType: string, paramName: string) => 
  SetMetadata(RESOURCE_OWNER_KEY, { resourceType, paramName });

// 2. 资源所有权守卫
// src/common/guards/resource-owner.guard.ts
import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { PrismaService } from '../../modules/prisma/prisma.service';

@Injectable()
export class ResourceOwnerGuard implements CanActivate {
  constructor(
    private reflector: Reflector,
    private prisma: PrismaService,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const resourceMeta = this.reflector.get<{
      resourceType: string;
      paramName: string;
    }>(RESOURCE_OWNER_KEY, context.getHandler());

    if (!resourceMeta) return true;

    const request = context.switchToHttp().getRequest();
    const userId = request.user?.id; // 从JWT中获取的当前用户ID
    const resourceId = request.params[resourceMeta.paramName];

    if (!userId || !resourceId) {
      throw new ForbiddenException('无法验证资源所有权');
    }

    const isOwner = await this.checkOwnership(
      resourceMeta.resourceType,
      resourceId,
      userId
    );

    if (!isOwner) {
      throw new ForbiddenException('您无权访问此资源');
    }

    return true;
  }

  private async checkOwnership(
    resourceType: string,
    resourceId: string,
    userId: string
  ): Promise<boolean> {
    const resourceOwners: Record<string, string> = {
      product: 'userId',
      order: 'userId',
      subscription: 'userId',
      aiUsageLog: 'userId',
    };

    const ownerField = resourceOwners[resourceType];
    if (!ownerField) return false;

    const count = await (this.prisma as any)[resourceType].count({
      where: {
        id: resourceId,
        [ownerField]: userId,
      },
    });

    return count > 0;
  }
}

// 3. 审计日志中间件
// src/common/interceptors/ownership-audit.interceptor.ts
import { Injectable, NestInterceptor, ExecutionContext, CallHandler } from '@nestjs/common';
import { Observable } from 'rxjs';
import { tap } from 'rxjs/operators';
import { PrismaService } from '../../modules/prisma/prisma.service';

@Injectable()
export class OwnershipAuditInterceptor implements NestInterceptor {
  constructor(private prisma: PrismaService) {}

  async intercept(context: ExecutionContext, next: CallHandler): Promise<Observable<any>> {
    const request = context.switchToHttp().getRequest();
    const { method, path, params, user, ip } = request;

    // 记录资源访问审计日志
    if (params.id && user?.id) {
      await this.prisma.auditLog.create({
        data: {
          userId: user.id,
          action: 'RESOURCE_ACCESS',
          resourceType: this.extractResourceType(path),
          resourceId: params.id,
          ip,
          userAgent: request.headers['user-agent'],
          timestamp: new Date(),
        },
      });
    }

    return next.handle();
  }

  private extractResourceType(path: string): string {
    const parts = path.split('/').filter(Boolean);
    return parts[0] || 'unknown';
  }
}

// src/modules/ai/input-filter.service.ts
@Injectable()
export class AiInputFilterService {
  private readonly suspiciousPatterns = [
    /system\s*:/i,
    /ignore\s+(previous|all)\s+(instructions|prompts)/i,
    /\[(system|admin|root)\]/i,
    /act\s+as\s+(root|admin|system)/i,
    /\\n\\n(?:system|user|assistant):/i,
  ];

  async filterInput(input: string): Promise<{
    passed: boolean;
    sanitized: string;
    reason?: string;
  }> {
    // 检查可疑模式
    for (const pattern of this.suspiciousPatterns) {
      if (pattern.test(input)) {
        return {
          passed: false,
          sanitized: input,
          reason: '输入包含可疑指令模式',
        };
      }
    }

    // 长度限制
    if (input.length > 10000) {
      return {
        passed: false,
        sanitized: input.substring(0, 10000),
        reason: '输入长度超过限制',
      };
    }

    return {
      passed: true,
      sanitized: this.escapeSpecialChars(input),
    };
  }

  private escapeSpecialChars(input: string): string {
    // 移除可能的注入字符
    return input
      .replace(/\\/g, '\\\\')
      .replace(/\[\s*(system|admin|root)\s*\]/gi, '[removed]');
  }
}

  private readonly sensitivePatterns = [
    /API[_-]?KEY/i,

    /SECRET/i,

    filtered: string;
    alerts: string[];

      if (pattern.test(output)) {
        alerts.push('检测到敏感信息模式');
        output = output.replace(pattern, '[REDACTED]');
      }
    }

    return {
      passed: alerts.length === 0,
      filtered: output,
      alerts,
    };
  }
}

  private readonly IDEMPOTENCY_TTL = 86400; // 24小时

  async check(idempotencyKey: string): Promise<{
    exists: boolean;
    status?: 'pending' | 'completed' | 'failed';
    result?: any;
  }> {
    const cached = await this.redis.get(`idempotent:${idempotencyKey}`);
    if (cached) {
      return JSON.parse(cached);
    }
    return { exists: false };
  }

  async markPending(idempotencyKey: string): Promise<void> {
    await this.redis.setex(
      `idempotent:${idempotencyKey}`,
      this.IDEMPOTENCY_TTL,
      JSON.stringify({ status: 'pending' })
    );
  }

  async markCompleted(idempotencyKey: string, result: any): Promise<void> {
    await this.redis.setex(
      `idempotent:${idempotencyKey}`,
      this.IDEMPOTENCY_TTL,
      JSON.stringify({ status: 'completed', result })
    );
  }

// src/common/encryption/encryption.service.ts
import * as crypto from 'crypto';

@Injectable()
export class EncryptionService {
  private readonly ALGORITHM = 'aes-256-gcm';
  private readonly KEY = Buffer.from(process.env.ENCRYPTION_KEY!, 'hex'); // 32字节

  // 加密
  encrypt(plaintext: string): EncryptedData {
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(this.ALGORITHM, this.KEY, iv);
    
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return {
      encrypted,
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex'),
    };
  }

  // 解密
  decrypt(data: EncryptedData): string {
    const decipher = crypto.createDecipheriv(
      this.ALGORITHM,
      this.KEY,
      Buffer.from(data.iv, 'hex')
    );
    decipher.setAuthTag(Buffer.from(data.authTag, 'hex'));
    
    let decrypted = decipher.update(data.encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }
}

// Prisma加密中间件
// src/prisma/encrypted-middleware.ts
export const encryptedFields = {
  user: ['idCardNumber', 'bankCardNumber'],
  order: ['shippingAddress'], // JSON字段需要整体加密
};

prisma.$use(async (params, next) => {
  if (params.model && encryptedFields[params.model as keyof typeof encryptedFields]) {
    const fields = encryptedFields[params.model as keyof typeof encryptedFields];
    
    // 写入时加密
    if (['create', 'update'].includes(params.action)) {
      const data = params.args.data;
      for (const field of fields) {
        if (data[field]) {
          data[field] = encrypt(JSON.stringify(data[field]));
        }
      }
    }
    
    // 读取时解密
    if (params.action === 'findUnique' || params.action === 'findFirst') {
      const result = await next(params);
      if (result) {
        for (const field of fields) {
          if (result[field]) {
            result[field] = JSON.parse(decrypt(result[field]));
          }
        }
      }
      return result;
    }
  }
  return next(params);
});

// src/modules/auth/refresh-token.service.ts
@Injectable()
export class RefreshTokenService {
  async generateTokenPair(userId: string): Promise<TokenPair> {
    const accessToken = await this.jwtService.signAsync(
      { sub: userId, type: 'access' },
      { expiresIn: '15m' }
    );
    
    const refreshToken = crypto.randomBytes(64).toString('hex');
    
    // 存储刷新Token（哈希存储）
    await this.redis.setex(
      `refresh:${this.hash(refreshToken)}`,
      7 * 24 * 3600, // 7天
      userId
    );
    
    // 记录Token版本（用于撤销）
    await this.redis.incr(`user:${userId}:token_version`);
    
    return { accessToken, refreshToken };
  }
  
  async refresh(refreshToken: string): Promise<TokenPair | null> {
    const tokenHash = this.hash(refreshToken);
    const userId = await this.redis.get(`refresh:${tokenHash}`);
    
    if (!userId) return null;
    
    // 撤销旧刷新Token
    await this.redis.del(`refresh:${tokenHash}`);
    
    // 生成新Token对
    return this.generateTokenPair(userId);
  }
  
  async revokeAllUserTokens(userId: string): Promise<void> {
    // 增加token版本号使所有旧Token失效
    await this.redis.incr(`user:${userId}:token_version`);
  }
}

    const pendingSecret = await this.redis.get(`mfa:pending:${userId}`);

    const verified = speakeasy.totp.verify({
      secret: pendingSecret,
      encoding: 'base32',

        where: { id: userId },
        data: { 
          mfaEnabled: true,
          mfaSecret: pendingSecret, // 应加密存储
        },
      });

@Injectable()
export class SecurityMonitorService {
  private readonly ALERT_THRESHOLDS = {
    loginAttempts: { count: 10, window: 300 }, // 5分钟内10次
    apiErrors: { rate: 0.1, window: 60 },       // 1分钟内10%错误率
    unusualAccess: { count: 100, window: 3600 }, // 1小时内100次访问
    dataExport: { count: 5, window: 3600 },       // 1小时内5次导出
  };
  
  // 异常登录检测
  async detectAnomalousLogin(userId: string): Promise<Alert | null> {
    const recentLogins = await this.redis.zrange(
      `login:history:${userId}`,
      0,
      -1
    );
    
    const loginLocations = recentLogins.map(l => JSON.parse(l).location);
    const uniqueLocations = new Set(loginLocations);
    
    // 检测短时间内多地登录
    if (uniqueLocations.size > 3) {
      return {
        type: 'SUSPICIOUS_LOGIN',
        severity: 'HIGH',
        userId,
        details: { locations: [...uniqueLocations] },
        recommendedAction: 'TEMP_LOCK_ACCOUNT',
      };
    }
    
    return null;
  }
  
  // API异常检测

    if (errorRate > this.ALERT_THRESHOLDS.apiErrors.rate) {
      return {
        type: 'HIGH_ERROR_RATE',
        severity: 'MEDIUM',
        details: { endpoint, errorRate },
        recommendedAction: 'INVESTIGATE',
      };
    }
    
    return null;
  }
